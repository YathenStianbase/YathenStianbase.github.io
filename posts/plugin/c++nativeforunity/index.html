C++ Native plugin for Unity [Unity C++原生插件管理]

2-minute read
Cplusplus
plugin

相关插件 UnityNativeTool

Unity提供的原生插件加载功能，在加载之后不能主动卸载，需要重新启动Unity，如果需要调试C++代码的话会非常的不方便;

UnityNativeTool可以很方便的在Editor Runtime期间进行C++ dll 的加载和卸载;

  • git clone 源码之后解压到Unity项目的Assets目录下；

    • 添加空物体，并挂载DllManipulatorScript脚本;

    • 启用"Enable in Edit Mode", 开启后会自动在程序退出时候卸载C++ dll;

接下来可以编写C++ 和 C#文件做交互了，一个简单的示例：

  • Visual Studio 2019 新建 C++ library 项目,并编写如下头文件和C++文件:

    • NativeTest.h 头文件

      #pragma once
#include <cstring>

extern "C" __declspec(dllexport) float NativeStart();
extern "C" __declspec(dllexport) float NativeEnd();

//编译器常量
extern "C" __declspec(dllexport) constexpr float ConstexprMethod();

extern "C" __declspec(dllexport) bool ReturnMyType();

class NativeClass
{
public:
    int i;
    float f;
    double d;
    char c;
    char* cp;
    NativeClass();
    ~NativeClass();

};

template<typename MyType>
struct my_type_struct {
    static constexpr bool boolval = false;
};

template<typename E>
constexpr bool has_flag(E lhs, E rhs)
{
    //return (lhs & rhs) == rhs;
    return (lhs == rhs);
}

    • NativeTest.cpp 文件

      #include "NativeTest.h"

NativeClass* nc;

float NativeStart()
{
    nc = new NativeClass();

    return nc == nullptr ? 0 : 3;
}

float NativeEnd()
{
    delete nc;
    return nc == NULL ? 0 : 1;
}

constexpr float ConstexprMethod()
{
    return 123.345f;
}

bool ReturnMyType() {
    return has_flag(1, 2);
    //return 2>1;
}

NativeClass::NativeClass() :i(0), f(10.0f), d(100.0), c('c'), cp(NULL)
{
}

NativeClass::~NativeClass()
{
    delete cp;
}

// 用做桥接的类，使用DLLImport导入C++ dll接口;
public class NativeDll
{
    // 指定的dll文件名称
    const string dllname = "NativeCode";
    public NativeDll(string _dllname)
    {
        // dllname = _dllname;
    }

    [DllImport(dllname)]
    public static extern float NativeStart();

    [DllImport(dllname)]
    public static extern float NativeEnd();

    [DllImport(dllname)]
    public static extern float ConstexprMethod();

    [DllImport(dllname)]
    public static extern bool ReturnMyType();
}
